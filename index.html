const express = require('express'); const bodyParser = require('body-parser'); const cors = require('cors'); require('dotenv').config(); const { Configuration, OpenAIApi } = require('openai'); const app = express(); app.use(cors()); app.use(bodyParser.json()); app.use(express.static(__dirname)); const configuration = new Configuration({ apiKey: process.env.OPENAI_API_KEY, }); const openai = new OpenAIApi(configuration); app.post('/chat', async (req, res) => { const userMessage = req.body.message; try { const response = await openai.createChatCompletion({ model: 'gpt-3.5-turbo', messages: [{ role: 'user', content: userMessage }], }); res.json({ reply: response.data.choices[0].message.content }); } catch (err) { res.status(500).json({ reply: 'שגיאה בשיחה: ' + err.message }); } }); app.listen(3000, () => console.log('השרת רץ בכתובת http://localhost:3000'));
